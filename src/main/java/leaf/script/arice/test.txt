c = 0; # 変数宣言は参照される行よりも前に書かなければならない

call b; # ルーチンは後ろの行から呼び出すこともできる

# ルーチンの定義
routine a{
	write(c);
	if(c==0|c>5){ # 条件分岐構文のテストです OR演算子が導入されました
		writeln("は0または5以上です");
		return; # return 文はルーチンから戻るときに使う
		write("ここは実行されない"); # 直前でreturnされるため実行されない
	}else{   # else if構文は定義されていません
		if((c%2)==0){
			writeln("は偶数です");
		}else{
			writeln("は奇数です");
		}
	}
	writeln("Hello, World"); #改行つきの画面出力
}
routine b{
	writeln("Yahooooooooo");
}
call b; # callはルーチンを呼び出します

while(c!=10){ # 繰り返し構文のテストです
	call a;
	c = c + 1;
}
call b;
